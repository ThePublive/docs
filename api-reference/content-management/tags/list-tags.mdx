title: "List Tags"
description: Fetch all tags
api: "GET https://cms.thepublive.com/publisher/{publisher_id}/tag/"

</ParamField>

<ParamField header="username" type="string" required>

</ParamField>

<ParamField header="password" type="string" required>
  API Secret
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number (max: 1000)
</ParamField>

<ParamField query="limit" type="integer" default="10">
  Items per page (max: 50)
</ParamField>
