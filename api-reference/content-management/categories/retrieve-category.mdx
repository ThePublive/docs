---
title: "Retrieve Category"
description: Retrieve a single category by ID
api: "GET https://cms.thepublive.com/publisher/{publisher_id}/category/{id}/"
---

</ParamField>

<ParamField path="id" type="integer" required>

</ParamField>

<ParamField header="username" type="string" required>
  API Key
</ParamField>

<ParamField header="password" type="string" required>
  API Secret
</ParamField>
